#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    long long D;
    cin >> N >> D;

    vector<long long> P(N);
    for (int i = 0; i < N; i++) {
        cin >> P[i];
    }

    sort(P.begin(), P.end(), greater<long long>());

    int wins = 0;
    int i = 0, j = N - 1;

    while (i <= j) {
        long long maxPower = P[i];
        long long need = D / maxPower + 1;

        if (i + need - 1 <= j) {
            wins++;
            i++;
            j -= (need - 1);
        } else {
            break;
        }
    }

    cout << wins;
    return 0;
}
